min_dist_ind=-1
min_dist=0
for i in range(len(centroids_selected)):
    dist = scipy.spatial.distance.cdist(centroids_selected.iloc[[i],:], new_sample, metric='euclidean')
    print(dist[0,0])
    if min_dist==0 or dist[0,0]<= min_dist:
        min_dist=dist[0,0]
        min_dist_ind=i
